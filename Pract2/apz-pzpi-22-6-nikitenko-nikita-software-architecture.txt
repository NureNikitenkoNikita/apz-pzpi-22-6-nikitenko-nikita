class MemoryManager {
    constructor(totalMemory) {
        this.totalMemory = totalMemory;
        this.allocatedMemory = 0;
        this.memoryMap = new Map(); // Map<processId, size>
        console.log(`–ú–µ–Ω–µ–¥–∂–µ—Ä –ø–∞–º'—è—Ç—ñ: –í—Å—å–æ–≥–æ ${totalMemory}MB.`);
    }

    // –Ü–º—ñ—Ç–∞—Ü—ñ—è –≤–∏–¥—ñ–ª–µ–Ω–Ω—è –ø–∞–º'—è—Ç—ñ –¥–ª—è –ø—Ä–æ—Ü–µ—Å—É
    allocate(processId, size) {
        if (this.allocatedMemory + size > this.totalMemory) {
            console.log(`–ü–∞–º'—è—Ç—å: –ù–µ–¥–æ—Å—Ç–∞—Ç–Ω—å–æ –ø–∞–º'—è—Ç—ñ –¥–ª—è –ø—Ä–æ—Ü–µ—Å—É ${processId} (${size}MB). –î–æ—Å—Ç—É–ø–Ω–æ: ${this.totalMemory - this.allocatedMemory}MB.`);
            return false;
        }
        this.memoryMap.set(processId, size);
        this.allocatedMemory += size;
        console.log(`–ü–∞–º'—è—Ç—å: –í–∏–¥—ñ–ª–µ–Ω–æ ${size}MB –¥–ª—è –ø—Ä–æ—Ü–µ—Å—É ${processId}. –ó–∞–π–Ω—è—Ç–æ: ${this.allocatedMemory}MB.`);
        return true;
    }

    // –Ü–º—ñ—Ç–∞—Ü—ñ—è –∑–≤—ñ–ª—å–Ω–µ–Ω–Ω—è –ø–∞–º'—è—Ç—ñ –ø—Ä–æ—Ü–µ—Å–æ–º
    deallocate(processId) {
        if (this.memoryMap.has(processId)) {
            const size = this.memoryMap.get(processId);
            this.allocatedMemory -= size;
            this.memoryMap.delete(processId);
            console.log(`–ü–∞–º'—è—Ç—å: –ó–≤—ñ–ª—å–Ω–µ–Ω–æ ${size}MB –≤—ñ–¥ –ø—Ä–æ—Ü–µ—Å—É ${processId}. –ó–∞–π–Ω—è—Ç–æ: ${this.allocatedMemory}MB.`);
            return true;
        }
        console.log(`–ü–∞–º'—è—Ç—å: –ü—Ä–æ—Ü–µ—Å ${processId} –Ω–µ –º–∞–≤ –≤–∏–¥—ñ–ª–µ–Ω–æ—ó –ø–∞–º'—è—Ç—ñ.`);
        return false;
    }
}

// VFS Core - —Ç–æ—á–∫–∞ –≤—Ö–æ–¥—É –¥–ª—è –∫–æ—Ä–∏—Å—Ç—É–≤–∞—á–∞
class VFS {
    constructor() {
        this.mountedFS = {}; // Map<mountPoint, FileSystemInterface>
    }

    mount(path, fsInstance) {
        this.mountedFS[path] = fsInstance;
        console.log(`VFS: –§–∞–π–ª–æ–≤–∞ —Å–∏—Å—Ç–µ–º–∞ –∑–º–æ–Ω—Ç–æ–≤–∞–Ω–∞ –≤ ${path}.`);
    }

    // –û–ø–µ—Ä–∞—Ü—ñ—ó, —è–∫—ñ –≤–∏–∫–ª–∏–∫–∞—î –∫–æ—Ä–∏—Å—Ç—É–≤–∞—á/–ø—Ä–æ–≥—Ä–∞–º–∞
    readFile(path) {
        for (const mountPoint in this.mountedFS) {
            if (path.startsWith(mountPoint)) {
                const relativePath = path;
                return this.mountedFS[mountPoint].read(relativePath);
            }
        }
        return "–ü–æ–º–∏–ª–∫–∞: –§–∞–π–ª–æ–≤–∞ —Å–∏—Å—Ç–µ–º–∞ –¥–ª—è —à–ª—è—Ö—É –Ω–µ –∑–Ω–∞–π–¥–µ–Ω–∞.";
    }
}

class KernelModule {
    constructor(name) {
        this.name = name;
        this.loaded = false;
    }

    // –§—É–Ω–∫—Ü—ñ—è, —â–æ –≤–∏–∫–æ–Ω—É—î—Ç—å—Å—è –ø—Ä–∏ –∑–∞–≤–∞–Ω—Ç–∞–∂–µ–Ω–Ω—ñ –º–æ–¥—É–ª—è
    init() {
        if (!this.loaded) {
            this.loaded = true;
            console.log(`–ú–æ–¥—É–ª—å –Ø–¥—Ä–∞: –ú–æ–¥—É–ª—å '${this.name}' –∑–∞–≤–∞–Ω—Ç–∞–∂–µ–Ω–æ.`);
        } else {
            console.log(`–ú–æ–¥—É–ª—å –Ø–¥—Ä–∞: –ú–æ–¥—É–ª—å '${this.name}' –≤–∂–µ –∑–∞–≤–∞–Ω—Ç–∞–∂–µ–Ω–∏–π.`);
        }
    }

    // –§—É–Ω–∫—Ü—ñ—è, —â–æ –≤–∏–∫–æ–Ω—É—î—Ç—å—Å—è –ø—Ä–∏ –≤–∏–≤–∞–Ω—Ç–∞–∂–µ–Ω–Ω—ñ –º–æ–¥—É–ª—è
    exit() {
        if (this.loaded) {
            this.loaded = false;
            console.log(`–ú–æ–¥—É–ª—å –Ø–¥—Ä–∞: –ú–æ–¥—É–ª—å '${this.name}' –≤–∏–≤–∞–Ω—Ç–∞–∂–µ–Ω–æ.`);
        } else {
            console.log(`–ú–æ–¥—É–ª—å –Ø–¥—Ä–∞: –ú–æ–¥—É–ª—å '${this.name}' –Ω–µ –∑–∞–≤–∞–Ω—Ç–∞–∂–µ–Ω–∏–π.`);
        }
    }

    // –î–æ–¥–∞—Ç–∫–æ–≤–∞ —Ñ—É–Ω–∫—Ü—ñ–æ–Ω–∞–ª—å–Ω—ñ—Å—Ç—å –º–æ–¥—É–ª—è
    performAction() {
        if (this.loaded) {
            console.log(`–ú–æ–¥—É–ª—å –Ø–¥—Ä–∞: –ú–æ–¥—É–ª—å '${this.name}' –≤–∏–∫–æ–Ω—É—î –¥—ñ—é.`);
        } else {
            console.log(`–ú–æ–¥—É–ª—å –Ø–¥—Ä–∞: –ú–æ–¥—É–ª—å '${this.name}' –Ω–µ–∞–∫—Ç–∏–≤–Ω–∏–π, –Ω–µ–º–æ–∂–ª–∏–≤–æ –≤–∏–∫–æ–Ω–∞—Ç–∏ –¥—ñ—é.`);
        }
    }
}

// –Ü–º—ñ—Ç–∞—Ü—ñ—è —è–¥—Ä–∞, —â–æ –∫–µ—Ä—É—î –º–æ–¥—É–ª—è–º–∏
class KernelModuleManager {
    constructor() {
        this.activeModules = new Map();
    }

    // –ö–æ–º–∞–Ω–¥–∞ 'insmod' (insert module)
    loadModule(moduleInstance) {
        if (!this.activeModules.has(moduleInstance.name)) {
            moduleInstance.init();
            this.activeModules.set(moduleInstance.name, moduleInstance);
        } else {
            console.log(`–ú–µ–Ω–µ–¥–∂–µ—Ä –ú–æ–¥—É–ª—ñ–≤: –ú–æ–¥—É–ª—å '${moduleInstance.name}' –≤–∂–µ –∑–∞–≤–∞–Ω—Ç–∞–∂–µ–Ω–∏–π.`);
        }
    }

    // –ö–æ–º–∞–Ω–¥–∞ 'rmmod' (remove module)
    unloadModule(moduleName) {
        if (this.activeModules.has(moduleName)) {
            const moduleInstance = this.activeModules.get(moduleName);
            moduleInstance.exit();
            this.activeModules.delete(moduleName);
        } else {
            console.log(`–ú–µ–Ω–µ–¥–∂–µ—Ä –ú–æ–¥—É–ª—ñ–≤: –ú–æ–¥—É–ª—å '${moduleName}' –Ω–µ –∑–Ω–∞–π–¥–µ–Ω–æ.`);
        }
    }
}
